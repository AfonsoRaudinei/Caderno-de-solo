class NovaAnaliseScreen extends ConsumerWidget {
  final AnaliseSolo? analiseParaEditar;

  const NovaAnaliseScreen({super.key, this.analiseParaEditar});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final ctrlProvider = novaAnaliseControllerProvider(analiseParaEditar);
    final state = ref.watch(ctrlProvider);
    final ctrl = ref.read(ctrlProvider.notifier);

    final derivados = ctrl.todosDerivados;

    return Scaffold(
      backgroundColor: AppColors.bgPrimary,
      appBar: AppBar(
        backgroundColor: AppColors.bgPrimary,
        elevation: 0,
        title: Text(
          analiseParaEditar == null ? 'Nova Análise' : 'Editar Análise',
          style: const TextStyle(
            fontSize: 17,
            fontWeight: FontWeight.w600,
            color: Color(0xFF1D1D1F),
          ),
        ),
        iconTheme: const IconThemeData(color: AppColors.primary),
        actions: [
          TextButton.icon(
            onPressed: state.isSaving
                ? null
                : () => _importarPdf(context, ctrl),
            icon: const Icon(Icons.upload_file, size: 18),
            label: const Text('Importar'),
            style: TextButton.styleFrom(
              foregroundColor: AppColors.primary,
              textStyle: const TextStyle(fontSize: 14),
            ),
          ),
        ],
      ),
      body: Column(
        children: [
          Expanded(
            child: SingleChildScrollView(
              padding: const EdgeInsets.only(bottom: 100),
              child: AnaliseTableWidget(
                analises: state.analises,
                derivados: derivados,
                onCampoChanged: ctrl.atualizarCampo,
                onGpsClicked: ctrl.capturarGps,
                onAddAnalise: ctrl.adicionarAnalise,
                onRemoveAnalise: ctrl.removerAnalise,
              ),
            ),
          ),
        ],
      ),
      floatingActionButtonLocation: FloatingActionButtonLocation.centerFloat,
      floatingActionButton: SizedBox(
        width: MediaQuery.of(context).size.width - 32,
        height: 52,
        child: ElevatedButton(
          onPressed: state.isSaving ? null : () => _salvar(context, ctrl),
          style: ElevatedButton.styleFrom(
            backgroundColor: AppColors.primary,
            foregroundColor: Colors.white,
            shape: RoundedRectangleBorder(
              borderRadius: BorderRadius.circular(12),
            ),
            elevation: 2,
          ),
          child: state.isSaving
              ? const SizedBox(
                  width: 20,
                  height: 20,
                  child: CircularProgressIndicator(
                      color: Colors.white, strokeWidth: 2),
                )
              : Text(
                  state.analises.length == 1
                      ? 'Salvar Análise'
                      : 'Salvar ${state.analises.length} Análises',
                  style: const TextStyle(
                      fontSize: 16, fontWeight: FontWeight.w600),
                ),
        ),
      ),
    );
  }

  Future<void> _salvar(
      BuildContext context, NovaAnaliseController ctrl) async {
    final ok = await ctrl.salvar('');
    if (!context.mounted) return;
    if (ok) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Análise salva com sucesso'),
          behavior: SnackBarBehavior.floating,
        ),
      );
      context.pop();
    } else {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Erro ao salvar — tente novamente'),
          behavior: SnackBarBehavior.floating,
          backgroundColor: AppColors.error,
        ),
      );
    }
  }

  Future<void> _importarPdf(
      BuildContext context, NovaAnaliseController ctrl) async {
    try {
      final analises =
          await const PdfImportService().importarDePdf(context);
      if (analises == null) return; // usuário cancelou

      ctrl.carregarDeImportacao(
        analises.map(_analiseSoloToMap).toList(),
      );

      if (!context.mounted) return;
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: Text(
              '${analises.length} amostra${analises.length > 1 ? 's' : ''} importada${analises.length > 1 ? 's' : ''} — revise e salve'),
          backgroundColor: AppColors.primary,
          behavior: SnackBarBehavior.floating,
          shape: RoundedRectangleBorder(
              borderRadius: BorderRadius.circular(10)),
        ),
      );
    } on LabNaoReconhecidoException {
      if (!context.mounted) return;
      showModalBottomSheet<void>(
        context: context,
        backgroundColor: Colors.transparent,
        builder: (_) => ImportacaoBottomSheet(
          tipo: ImportacaoBottomSheetTipo.labNaoReconhecido,
          onDigitarManualmente: () => Navigator.of(context).pop(),
        ),
      );
    } on ExtracacaoIndisponivelException {
      if (!context.mounted) return;
      showModalBottomSheet<void>(
        context: context,
        backgroundColor: Colors.transparent,
        builder: (_) => ImportacaoBottomSheet(
          tipo: ImportacaoBottomSheetTipo.extracacaoIndisponivel,
          onDigitarManualmente: () => Navigator.of(context).pop(),
        ),
      );
    } catch (e) {
      if (!context.mounted) return;
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Erro ao importar: $e')),
      );
    }
  }

### 3. Localizar o controller da tela
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:soloforte/core/services/location_service.dart';
import 'package:soloforte/features/analise/domain/entities/analise_solo.dart';

/// Estado do controller da tabela de nova análise
class NovaAnaliseState {
  final List<Map<String, dynamic>> analises;
  final bool isSaving;
  final String? error;

  const NovaAnaliseState({
    required this.analises,
    this.isSaving = false,
    this.error,
  });

  NovaAnaliseState copyWith({
    List<Map<String, dynamic>>? analises,
    bool? isSaving,
    String? error,
  }) {
    return NovaAnaliseState(
      analises: analises ?? this.analises,
      isSaving: isSaving ?? this.isSaving,
      error: error,
    );
  }
}

/// Controller para a tela NovaAnalise em formato tabela planilha.
class NovaAnaliseController extends StateNotifier<NovaAnaliseState> {
  final Ref _ref;
  final _calc = const CalcularDerivadosAnalise();
  final _uuid = const Uuid();

  /// Máximo de colunas (amostras) permitido
  static const int maxColunas = 6;

  NovaAnaliseController(this._ref, {AnaliseSolo? analiseParaEditar})
      : super(
          NovaAnaliseState(
            analises: analiseParaEditar != null
                ? [_soloToMap(analiseParaEditar)]
                : [{}],
          ),
        );

  // ── Conversão entidade → map ────────────────────────────────────────────

  static Map<String, dynamic> _soloToMap(AnaliseSolo a) {
    return {
      'fazenda': a.fazenda,
      'produtor': a.produtor,
      'talhao': a.talhao,
      'numeroAmostra': a.numeroAmostra,
      'cultura': a.cultura.label,
      'safra': a.safra,
      'laboratorio': a.laboratorio,
      'profundidade': a.profundidade,
      'latitude': a.latitude?.toString() ?? '',
      'longitude': a.longitude?.toString() ?? '',
      'descricaoLocal': a.descricaoLocal ?? '',
      'pdfUrl': a.pdfUrl ?? '',
      'argila': a.argila?.toString() ?? '',
      'silte': a.silte?.toString() ?? '',
      'areiaTotal': a.areiaTotal?.toString() ?? '',
      'phAgua': a.phAgua?.toString() ?? '',
      'phCaCl2': a.phCaCl2?.toString() ?? '',
      'phSmp': a.phSmp?.toString() ?? '',
      'materiaOrganica': a.materiaOrganica?.toString() ?? '',
      'carbonoOrganico': a.carbonoOrganico?.toString() ?? '',
      'pMehlich': a.pMehlich?.toString() ?? '',
      'pResina': a.pResina?.toString() ?? '',
      'pRem': a.pRem?.toString() ?? '',
      's020': a.s020?.toString() ?? '',
      's2040': a.s2040?.toString() ?? '',
      'k': a.k?.toString() ?? '',
      'ca': a.ca?.toString() ?? '',
      'mg': a.mg?.toString() ?? '',
      'al': a.al?.toString() ?? '',
      'hMaisAl': a.hMaisAl?.toString() ?? '',
      'na': a.na?.toString() ?? '',
      'b': a.b?.toString() ?? '',
      'cu': a.cu?.toString() ?? '',
      'fe': a.fe?.toString() ?? '',
      'mn': a.mn?.toString() ?? '',
      'zn': a.zn?.toString() ?? '',
      'ni': a.ni?.toString() ?? '',
      'mo': a.mo?.toString() ?? '',
      'se': a.se?.toString() ?? '',
    };
  }

  // ── Gerência de colunas ────────────────────────────────────────────────

  void adicionarAnalise() {
    if (state.analises.length >= maxColunas) return;
    state = state.copyWith(analises: [...state.analises, {}]);
  }

  void removerAnalise(int index) {
    if (state.analises.length <= 1) return;
    final updated = [...state.analises]..removeAt(index);
    state = state.copyWith(analises: updated);
  }

  void atualizarCampo(int index, String campo, dynamic valor) {
    final updated = [...state.analises];
    updated[index] = {...updated[index], campo: valor};
    state = state.copyWith(analises: updated);
  }

  Future<void> capturarGps(int analiseIndex) async {
    final location = await LocationService().getCurrentLocation();
    if (location == null) return;
    
    atualizarCampo(analiseIndex, 'latitude', location.latitude.toString());
    atualizarCampo(analiseIndex, 'longitude', location.longitude.toString());
  }

  // ── Derivados ──────────────────────────────────────────────────────────

  Map<String, double> calcularDerivados(Map<String, dynamic> analise) {
    double? pd(String key) {
      final v = analise[key];
      if (v == null) return null;
      if (v is double) return v;
      if (v is num) return v.toDouble();
      if (v is String) return double.tryParse(v.replaceAll(',', '.'));
      return null;
    }

    return _calc.call({
      'ca': pd('ca'),
      'mg': pd('mg'),
      'k': pd('k'),
      'na': pd('na') ?? 0.0,
      'al': pd('al'),
      'hMaisAl': pd('hMaisAl'),
    });
  }

  List<Map<String, double>> get todosDerivados =>
      state.analises.map(calcularDerivados).toList();

  // ── Importação ─────────────────────────────────────────────────────────

  /// Carrega dados importados de um laboratório.
  /// Cada item da lista é um mapa com chaves canônicas.
  void carregarDeImportacao(List<Map<String, dynamic>> dados) {
    if (dados.isEmpty) return;
    final maps = dados.map((d) {
      return d.map((k, v) => MapEntry(k, v?.toString() ?? ''));
    }).toList();
    state = state.copyWith(analises: maps);
  }

  /// Converte uma [AnaliseSolo] importada para o mapa usado pelo controller.
  void carregarDeAnaliseSolo(List<AnaliseSolo> analises) {
    if (analises.isEmpty) return;
    final maps = analises.map(_soloToMap).toList();
    state = state.copyWith(analises: maps);
  }

  // ── Persistência ───────────────────────────────────────────────────────

  Future<bool> salvar(String produtorPadrao) async {
    state = state.copyWith(isSaving: true);
    try {
      final notifier = _ref.read(analiseNotifierProvider.notifier);
      for (final map in state.analises) {
        final analise = _mapToAnaliseSolo(map, produtorPadrao);
        await notifier.salvar(analise);
      }
      state = state.copyWith(isSaving: false);
      return true;
    } catch (e) {
      state = state.copyWith(isSaving: false, error: e.toString());
      return false;
    }
  }

  AnaliseSolo _mapToAnaliseSolo(
      Map<String, dynamic> map, String produtorPadrao) {
    String s(String key, [String fallback = '']) =>
        (map[key] as String?)?.trim() ?? fallback;

    double? d(String key) {
      final v = map[key];
      if (v == null || (v is String && v.trim().isEmpty)) return null;
      if (v is num) return v.toDouble();
      return double.tryParse(v.toString().replaceAll(',', '.'));
    }

    Cultura parseCultura(String raw) {
      final normalized = raw.trim().toLowerCase();
      for (final c in Cultura.values) {
        if (c.name == normalized || c.label.toLowerCase() == normalized) {
          return c;
        }
      }
      return Cultura.soja;
    }

    return AnaliseSolo(
      id: _uuid.v4(),
      fazenda: s('fazenda'),
      produtor: s('produtor', produtorPadrao),
      talhao: s('talhao'),
      numeroAmostra: s('numeroAmostra'),
      cultura: parseCultura(s('cultura')),
      safra: s('safra'),
      laboratorio: s('laboratorio'),
      dataCadastro: DateTime.now(),
      profundidade: s('profundidade'),
      latitude: d('latitude'),
      longitude: d('longitude'),
      descricaoLocal: s('descricaoLocal'),
      pdfUrl: s('pdfUrl'),
      argila: d('argila'),
      silte: d('silte'),
      areiaTotal: d('areiaTotal'),
      phAgua: d('phAgua'),
      phCaCl2: d('phCaCl2'),
      phSmp: d('phSmp'),
      materiaOrganica: d('materiaOrganica'),
      carbonoOrganico: d('carbonoOrganico'),
      pMehlich: d('pMehlich'),
      pResina: d('pResina'),
      pRem: d('pRem'),
      s020: d('s020'),
      s2040: d('s2040'),
      k: d('k'),
      ca: d('ca'),
      mg: d('mg'),
      al: d('al'),
      hMaisAl: d('hMaisAl'),
      na: d('na'),
      b: d('b'),
      cu: d('cu'),
      fe: d('fe'),
      mn: d('mn'),
      zn: d('zn'),
      ni: d('ni'),
      mo: d('mo'),
      se: d('se'),
    );
  }
}

// ── Provider ────────────────────────────────────────────────────────────────

final novaAnaliseControllerProvider = StateNotifierProvider.family
    .autoDispose<NovaAnaliseController, NovaAnaliseState, AnaliseSolo?>(
  (ref, analiseParaEditar) =>
      NovaAnaliseController(ref, analiseParaEditar: analiseParaEditar),
);
